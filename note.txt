CREATE DATABASE fz

CREATE TABLE `pengguna` (		
    `id_pengguna` int(11) NOT NULL AUTO_INCREMENT,
    `name` varchar(100) DEFAULT NULL,
    `username` varchar(100) DEFAULT NULL,
    `password` varchar(100) DEFAULT NULL,
    `hakakses` int(11) DEFAULT NULL,
    `status` varchar(10) DEFAULT NULL,
    PRIMARY KEY (`id_pengguna`)
)

CREATE TABLE `alat_berat` (	
    `id_alat_berat` int(11) NOT NULL AUTO_INCREMENT,
    nama_alat_berat varchar(100) DEFAULT NULL,
    tipe_alat_berat varchar(100) DEFAULT NULL,
    deskripsi text DEFAULT NULL,
    tahun_pembuatan YEAR(4) DEFAULT NULL,
    PRIMARY KEY (`id_alat_berat`)
)

CREATE TABLE `perawatan` (
    id_perawatan int(11) NOT NULL AUTO_INCREMENT,
    id_alat_berat int(11) NOT NULL,
    id_pengguna int(11) NOT NULL,
    tanggal_perawatan date DEFAULT NULL,
    jenis_perawatan varchar(100) DEFAULT NULL,
    keterangan varchar(225) DEFAULT NULL,
    biaya_perawatan int(11) DEFAULT NULL,
    PRIMARY KEY (`id_perawatan`)
)

CREATE TABLE `komponen` (
    id_komponen int(11) NOT NULL AUTO_INCREMENT,
    id_alat_berat int(11) NOT NULL,
    nama_komponen varchar(100) DEFAULT NULL,
    deskripsi text DEFAULT NULL,
    tipe_komponen varchar(100) DEFAULT NULL,
    tahun_pembuatan date DEFAULT NULL,
    PRIMARY KEY (`id_komponen`)
)

CREATE TABLE `kriteria` (
    id_kriteria int(11) NOT NULL AUTO_INCREMENT,
    nama_kriteria varchar(100) DEFAULT NULL,
    deskripsi text DEFAULT NULL,
    PRIMARY KEY (`id_kriteria`)
)

CREATE TABLE `sub_kriteria` (
    id_sub_kriteria int(11) NOT NULL AUTO_INCREMENT,
    id_kriteria int(11) NOT NULL,
    sub_kriteria varchar(100) DEFAULT NULL,
    deskripsi text DEFAULT NULL,
    PRIMARY KEY (`id_sub_kriteria`)
)

CREATE TABLE `perbandingan` (
    id_perbandingan  int(11) NOT NULL AUTO_INCREMENT,
    sub_kriteria1 int(11) NOT NULL,
    sub_kriteria2 int(11) NOT NULL,
    nilai int(11) DEFAULT NULL,
    PRIMARY KEY (`id_perbandingan`)
)
